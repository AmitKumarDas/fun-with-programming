func ProdExceptSelf(given []int) []int {
  var prod int = 1
  for _, i := range given {
    prod *= i
  }
  
  // -------------
  // NOT USING MAKE SEEMS BETTER FOR SMALL ARRAY
  // -------------
  
  // var op = make([]int, len(given))
  var op []int
  
  for _, i := range given {
    val := int(prod/i)
    op = append(op, val)
  }
  return op
}
