// O(1) space
// O(n^2) runtime - BAD

func DeleteDuplicates(sorted []int) []int {
  for j := len(sorted)-1; j > 0; j-- {
    if sorted[j] != sorted[j-1] {
      continue
    }
    tmp := j
    for tmp < len(sorted) {
      var rightVal
      if tmp+1 < len(sorted) {
        rightVal = sorted[tmp+1]
      }
      sorted[tmp]=rightVal
      tmp++
    }
  }
  return sorted
}
